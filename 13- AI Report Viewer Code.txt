import React, { useState } from 'react';
import ReactMarkdown from 'react-markdown';
import { 
  FileText, 
  Copy, 
  Download, 
  Check, 
  Table as TableIcon, 
  FileSpreadsheet,
  Printer,
  Maximize2
} from 'lucide-react';

/**
 * AIReportViewer
 * Renders rich Markdown reports from the 'ai_reports' table.
 * Specifically optimized for high-density B2B tables and data extraction results.
 */

interface AIReportViewerProps {
  report?: {
    title: string;
    content_md: string;
    data_payload: any[];
    created_at: string;
  };
  isLoading?: boolean;
}

export default function AIReportViewer({ report, isLoading }: AIReportViewerProps) {
  const [copied, setCopied] = useState(false);

  // Default data structure if none provided
  const activeReport = report || {
    title: "Structural Material Estimate - Project Alpha",
    created_at: new Date().toISOString(),
    content_md: `
### Executive Summary
The AI has analyzed the provided CAD floorplans (Revision 4). The following material quantities have been extracted with an overall confidence score of **92%**.

#### Material Breakdown
* **Concrete Slab**: Grade 30, 200mm | 450.5 m³ | 94% Confidence
* **Steel Reinforcement**: 12mm Rebar | 12.4 tonnes | 82% Confidence
* **Formwork**: Timber | 890.0 m² | 65% Confidence
* **Vapor Barrier**: High Density | 510.0 m² | 98% Confidence

> **Note:** Low confidence in Formwork calculation due to overlapping layers in Section B-12. Manual verification recommended.
    `,
    data_payload: [
      { item: "Concrete Slab", spec: "Grade 30, 200mm", qty: 450.5, unit: "m³", conf: 0.94 },
      { item: "Steel Reinforcement", spec: "12mm Rebar", qty: 12.4, unit: "tonnes", conf: 0.82 },
      { item: "Formwork", spec: "Timber", qty: 890.0, unit: "m²", conf: 0.65 },
      { item: "Vapor Barrier", spec: "High Density", qty: 510.0, unit: "m²", conf: 0.98 }
    ]
  };

  const handleCopyTable = () => {
    // Generate tab-separated string for easy pasting into Excel/Sheets
    const headers = "Material Item\tSpecification\tQuantity\tUnit\tConfidence\n";
    const rows = activeReport.data_payload.map((r: any) => 
      `${r.item}\t${r.spec}\t${r.qty}\t${r.unit}\t${(r.conf * 100).toFixed(0)}%`
    ).join('\n');
    
    // Fallback for clipboard copy in restricted environments
    const textArea = document.createElement("textarea");
    textArea.value = headers + rows;
    document.body.appendChild(textArea);
    textArea.select();
    document.execCommand('copy');
    document.body.removeChild(textArea);
    
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };

  const downloadCSV = () => {
    const headers = "Material Item,Specification,Quantity,Unit,Confidence\n";
    const rows = activeReport.data_payload.map((r: any) => 
      `"${r.item}","${r.spec}",${r.qty},"${r.unit}",${(r.conf * 100).toFixed(0)}%`
    ).join('\n');
    const blob = new Blob([headers + rows], { type: 'text/csv' });
    const url = window.URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.setAttribute('hidden', '');
    a.setAttribute('href', url);
    a.setAttribute('download', 'ai_report_export.csv');
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
  };

  if (isLoading) {
    return (
      <div className="w-full h-96 flex flex-col items-center justify-center space-y-4 bg-muted/10 rounded-xl border border-dashed">
        <div className="w-12 h-12 border-4 border-blue-500 border-t-transparent rounded-full animate-spin" />
        <p className="text-muted-foreground animate-pulse font-medium text-sm">Generating Final Report...</p>
      </div>
    );
  }

  return (
    <div className="flex flex-col h-full bg-background border rounded-xl shadow-lg overflow-hidden w-full max-w-5xl mx-auto">
      {/* TOOLBAR */}
      <div className="flex flex-wrap items-center justify-between gap-4 px-6 py-4 border-b bg-muted/20">
        <div className="flex items-center gap-3">
          <div className="p-2 bg-blue-600 text-white rounded-lg shadow-sm">
            <FileText className="w-5 h-5" />
          </div>
          <div>
            <h1 className="text-base font-bold leading-none">{activeReport.title}</h1>
            <p className="text-[10px] text-muted-foreground mt-1 uppercase tracking-wider">
              Analysis Date: {new Date(activeReport.created_at).toLocaleDateString()}
            </p>
          </div>
        </div>

        <div className="flex items-center gap-2">
          <button 
            onClick={handleCopyTable}
            className="flex items-center gap-2 px-3 py-1.5 text-xs font-semibold bg-background border rounded-lg hover:bg-muted transition-all active:scale-95"
          >
            {copied ? <Check className="w-3.5 h-3.5 text-green-600" /> : <Copy className="w-3.5 h-3.5" />}
            {copied ? 'Copied' : 'Excel Copy'}
          </button>
          
          <button 
            onClick={downloadCSV}
            className="flex items-center gap-2 px-3 py-1.5 text-xs font-semibold bg-blue-600 text-white rounded-lg hover:bg-blue-700 transition-all shadow-sm"
          >
            <Download className="w-3.5 h-3.5" />
            Export CSV
          </button>

          <div className="w-px h-6 bg-border mx-1" />

          <button className="p-2 text-muted-foreground hover:bg-muted rounded-lg transition-colors">
            <Printer className="w-4 h-4" />
          </button>
        </div>
      </div>

      {/* MAIN CONTENT AREA */}
      <div className="flex-1 overflow-y-auto p-8 prose prose-slate max-w-none dark:prose-invert prose-headings:font-bold prose-blockquote:border-l-4 prose-blockquote:border-blue-500 prose-blockquote:bg-blue-50/50 prose-blockquote:p-4 prose-blockquote:rounded-r-lg prose-blockquote:italic">
        <ReactMarkdown 
          components={{
            blockquote: ({node, ...props}) => (
              <blockquote className="border-l-4 border-blue-500 bg-blue-50/50 p-4 rounded-r-lg italic" {...props} />
            )
          }}
        >
          {activeReport.content_md}
        </ReactMarkdown>

        {/* HIGH-DENSITY B2B DATA TABLE */}
        {activeReport.data_payload && activeReport.data_payload.length > 0 && (
          <div className="my-8 overflow-x-auto rounded-lg border shadow-sm bg-background">
            <table className="min-w-full divide-y divide-border m-0 text-xs">
              <thead className="bg-muted/50">
                <tr>
                  <th className="px-4 py-3 text-left font-bold text-foreground">Material Item</th>
                  <th className="px-4 py-3 text-left font-bold text-foreground">Specification</th>
                  <th className="px-4 py-3 text-right font-bold text-foreground">Quantity</th>
                  <th className="px-4 py-3 text-left font-bold text-foreground">Unit</th>
                  <th className="px-4 py-3 text-center font-bold text-foreground">Confidence</th>
                </tr>
              </thead>
              <tbody className="divide-y divide-border">
                {activeReport.data_payload.map((row: any, i: number) => (
                  <tr key={i} className="hover:bg-muted/30 transition-colors">
                    <td className="px-4 py-3 font-medium">{row.item}</td>
                    <td className="px-4 py-3 text-muted-foreground">{row.spec}</td>
                    <td className="px-4 py-3 text-right font-mono">{row.qty}</td>
                    <td className="px-4 py-3 text-muted-foreground">{row.unit}</td>
                    <td className="px-4 py-3 text-center">
                      <span className={`inline-block px-2 py-0.5 rounded-[4px] text-[10px] font-bold ${
                        row.conf > 0.9 ? 'bg-green-100 text-green-700' : 
                        row.conf > 0.7 ? 'bg-blue-100 text-blue-700' : 
                        'bg-amber-100 text-amber-700'
                      }`}>
                        {(row.conf * 100).toFixed(0)}%
                      </span>
                    </td>
                  </tr>
                ))}
              </tbody>
            </table>
          </div>
        )}
      </div>

      {/* SYSTEM STATUS FOOTER */}
      <div className="px-6 py-3 border-t bg-muted/10 flex items-center justify-between text-[10px] text-muted-foreground uppercase tracking-widest font-bold">
        <div className="flex items-center gap-4">
          <span className="flex items-center gap-1.5">
            <span className="w-2 h-2 rounded-full bg-green-500 animate-pulse" />
            AI Verified
          </span>
          <span className="hidden sm:inline">SHA-256: 4F92A...E81D</span>
        </div>
        <div className="flex items-center gap-4">
          <span className="flex items-center gap-1">
            <TableIcon className="w-3 h-3" /> Grid Ready
          </span>
          <span className="flex items-center gap-1">
            <FileSpreadsheet className="w-3 h-3" /> Excel Sync Enabled
          </span>
        </div>
      </div>
    </div>
  );
}
