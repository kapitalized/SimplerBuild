import React, { useState, useEffect } from 'react';
import { 
  CheckCircle2, 
  CircleDashed, 
  AlertCircle, 
  Loader2, 
  ChevronRight, 
  FileText, 
  Layers, 
  Calculator,
  ExternalLink,
  ShieldCheck
} from 'lucide-react';

/**
 * AITaskStatus
 * * High-fidelity task tracker for the B2B AI pipeline.
 * * Handles the transition between Extraction, Analysis, and Synthesis.
 * * Features simulated state transitions for demonstration/prototyping.
 */

type TaskStep = {
  id: string;
  label: string;
  status: 'pending' | 'processing' | 'completed' | 'failed';
  icon: React.ReactNode;
  description: string;
};

type AIJobStatus = 'idle' | 'running' | 'review' | 'completed' | 'failed';

export default function AITaskStatus({ 
  jobId = "task_88231", 
  onVerify 
}: { 
  jobId?: string, 
  onVerify?: (data: any) => void 
}) {
  const [jobStatus, setJobStatus] = useState<AIJobStatus>('running');
  const [activeStep, setActiveStep] = useState(0);

  // MOCK STEPS for a CAD Material Quantity Estimate or Financial Audit
  const [steps, setSteps] = useState<TaskStep[]>([
    { 
      id: 'ingest', 
      label: 'File Ingestion', 
      status: 'completed', 
      icon: <FileText className="w-4 h-4" />,
      description: 'Document converted to high-res vision tensors.'
    },
    { 
      id: 'extract', 
      label: 'Visual Extraction', 
      status: 'processing', 
      icon: <Layers className="w-4 h-4" />,
      description: 'Gemini 2.0 Flash identifying data layers and citations...'
    },
    { 
      id: 'analyze', 
      label: 'Quantity Analysis', 
      status: 'pending', 
      icon: <Calculator className="w-4 h-4" />,
      description: 'DeepSeek-R1 calculating volumes and normalizing data.'
    },
  ]);

  /**
   * SIMULATION LOGIC
   * This simulates the transition from 'Extraction' to 'Analysis' to 'Human Review'
   */
  useEffect(() => {
    if (jobStatus !== 'running') return;

    const timer = setTimeout(() => {
      if (activeStep === 1) {
        // Complete Step 2, Start Step 3
        const newSteps = [...steps];
        newSteps[1].status = 'completed';
        newSteps[2].status = 'processing';
        setSteps(newSteps);
        setActiveStep(2);
      } else if (activeStep === 2) {
        // Complete Step 3, Move to Review
        const newSteps = [...steps];
        newSteps[2].status = 'completed';
        setSteps(newSteps);
        setJobStatus('review');
      }
    }, 3500);

    return () => clearTimeout(timer);
  }, [activeStep, jobStatus, steps]);

  return (
    <div className="max-w-md w-full bg-card border rounded-xl shadow-lg overflow-hidden flex flex-col font-sans text-sm">
      {/* HEADER */}
      <div className="p-4 border-b bg-muted/30 flex items-center justify-between">
        <div className="flex items-center gap-2">
          <div className={`p-1.5 rounded-md ${jobStatus === 'running' ? 'bg-blue-100 text-blue-600 animate-pulse' : 'bg-green-100 text-green-600'}`}>
            {jobStatus === 'running' ? <CircleDashed className="w-4 h-4 animate-spin-slow" /> : <ShieldCheck className="w-4 h-4" />}
          </div>
          <div>
            <h3 className="font-semibold leading-none">AI Intelligence Layer</h3>
            <p className="text-xs text-muted-foreground mt-1 uppercase tracking-wider font-bold">Job ID: {jobId}</p>
          </div>
        </div>
        <div className="text-[10px] font-mono bg-background border px-2 py-1 rounded font-bold">
          {jobStatus === 'running' ? 'EXECUTING' : jobStatus.toUpperCase()}
        </div>
      </div>

      {/* STEPS LIST */}
      <div className="p-5 space-y-6">
        {steps.map((step, index) => (
          <div key={step.id} className="flex gap-4 group">
            <div className="flex flex-col items-center">
              <div className={`
                z-10 flex items-center justify-center w-8 h-8 rounded-full border-2 transition-all duration-300
                ${step.status === 'completed' ? 'bg-green-500 border-green-500 text-white shadow-md' : 
                  step.status === 'processing' ? 'bg-background border-blue-500 text-blue-500 shadow-[0_0_12px_rgba(59,130,246,0.3)]' : 
                  'bg-background border-muted text-muted-foreground'}
              `}>
                {step.status === 'completed' ? <CheckCircle2 className="w-5 h-5" /> : 
                 step.status === 'processing' ? <Loader2 className="w-4 h-4 animate-spin text-blue-500" /> : 
                 <span className="text-xs font-bold">{index + 1}</span>}
              </div>
              {index !== steps.length - 1 && (
                <div className={`w-0.5 h-full mt-2 rounded-full transition-colors duration-500 ${step.status === 'completed' ? 'bg-green-500' : 'bg-muted'}`} />
              )}
            </div>
            <div className="pb-2 flex-1">
              <div className="flex items-center justify-between">
                <span className={`font-bold ${step.status === 'processing' ? 'text-blue-600' : 'text-foreground'}`}>
                  {step.label}
                </span>
                <span className="text-[10px] text-muted-foreground font-mono">
                  {step.status === 'completed' ? 'DONE' : step.status === 'processing' ? 'LIVE' : '--'}
                </span>
              </div>
              <p className="text-muted-foreground text-xs mt-1 leading-relaxed">
                {step.description}
              </p>
            </div>
          </div>
        ))}
      </div>

      {/* FOOTER / CALL TO ACTION */}
      <div className={`p-4 border-t transition-all duration-500 ${jobStatus === 'review' ? 'bg-blue-50/50' : 'bg-muted/10'}`}>
        {jobStatus === 'running' ? (
          <div className="flex items-center justify-center gap-2 text-muted-foreground animate-pulse py-1">
            <Loader2 className="w-4 h-4 animate-spin" />
            <span className="text-xs font-bold uppercase tracking-widest">Orchestrating Pipeline...</span>
          </div>
        ) : jobStatus === 'review' ? (
          <div className="space-y-3">
            <div className="flex items-start gap-2 text-blue-700">
              <AlertCircle className="w-4 h-4 mt-0.5 shrink-0" />
              <p className="text-[11px] leading-tight font-medium">
                <strong>Review Required:</strong> High-density data extraction complete. Please verify citations before final export.
              </p>
            </div>
            <button 
              onClick={() => { setJobStatus('completed'); onVerify?.({}); }}
              className="w-full py-2.5 bg-blue-600 hover:bg-blue-700 text-white rounded-lg font-bold transition-all shadow-md flex items-center justify-center gap-2 active:scale-95"
            >
              Verify Results <ChevronRight className="w-4 h-4" />
            </button>
          </div>
        ) : (
          <div className="flex items-center justify-between">
            <div className="flex items-center gap-2 text-green-700 font-bold">
              <CheckCircle2 className="w-4 h-4" />
              <span>Analysis Finalized</span>
            </div>
            <button className="text-blue-600 hover:underline text-xs flex items-center gap-1 font-bold">
              View Final Report <ExternalLink className="w-3 h-3" />
            </button>
          </div>
        )}
      </div>

      <style>{`
        @keyframes spin-slow {
          from { transform: rotate(0deg); }
          to { transform: rotate(360deg); }
        }
        .animate-spin-slow {
          animation: spin-slow 3s linear infinite;
        }
      `}</style>
    </div>
  );
}
