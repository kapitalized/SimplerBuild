import { CollectionConfig } from 'payload/types';
import crypto from 'crypto';

/**
 * 18 - Payload Encryption Hook
 * Implementation of the AES-256-GCM "Vault Pattern" for Payload CMS.
 * Requires ENCRYPTION_KEY in .env (Must be exactly 32 characters).
 */

const ENCRYPTION_KEY = process.env.ENCRYPTION_KEY; 
const IV_LENGTH = 16;
const AUTH_TAG_LENGTH = 16;

/**
 * encrypt
 * Converts plain-text API keys into a secure iv:tag:ciphertext format.
 */
const encrypt = (text: string): string => {
  if (!ENCRYPTION_KEY || ENCRYPTION_KEY.length !== 32) {
    throw new Error('ENCRYPTION_KEY must be exactly 32 characters long for AES-256.');
  }

  const iv = crypto.randomBytes(IV_LENGTH);
  const cipher = crypto.createCipheriv('aes-256-gcm', Buffer.from(ENCRYPTION_KEY), iv);
  
  const encrypted = Buffer.concat([cipher.update(text, 'utf8'), cipher.final()]);
  const tag = cipher.getAuthTag();

  // Store in database as iv:tag:encryptedData
  return `${iv.toString('hex')}:${tag.toString('hex')}:${encrypted.toString('hex')}`;
};

const ExternalIntegrations: CollectionConfig = {
  slug: 'external-integrations',
  admin: {
    useAsTitle: 'serviceName',
    group: 'Admin',
    description: 'Manage 3rd-party API credentials (Xero, Stripe, ERP) for your clients.',
  },
  access: {
    // Only super-admins can interact with these secrets directly
    read: ({ req: { user } }) => user?.role === 'admin',
    update: ({ req: { user } }) => user?.role === 'admin',
    create: ({ req: { user } }) => user?.role === 'admin',
  },
  fields: [
    {
      name: 'serviceName',
      type: 'text',
      required: true,
      admin: {
        placeholder: 'e.g., Client Xero Integration',
      }
    },
    {
      name: 'environment',
      type: 'select',
      defaultValue: 'sandbox',
      options: [
        { label: 'Sandbox / Testing', value: 'sandbox' },
        { label: 'Production / Live', value: 'production' },
      ],
    },
    {
      name: 'apiKey',
      type: 'text',
      required: true,
      admin: {
        description: 'Paste the plain-text secret here. It will be encrypted immediately upon saving.',
      },
      hooks: {
        beforeChange: [
          ({ value, operation }) => {
            if (operation === 'create' || operation === 'update') {
              // Only encrypt if it's not already in the encrypted format (iv:tag:data)
              if (value && !value.includes(':')) { 
                return encrypt(value);
              }
            }
            return value;
          },
        ],
      },
    },
  ],
};

export default ExternalIntegrations;
