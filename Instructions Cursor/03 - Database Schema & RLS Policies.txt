-- ENABLE EXTENSIONS
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
CREATE EXTENSION IF NOT EXISTS vector; -- Required for RAG [cite: 19]

-- 1. ORGANIZATIONS (The Multi-tenant Root)
CREATE TABLE organizations (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name TEXT NOT NULL,
  stripe_customer_id TEXT,
  stripe_subscription_id TEXT,
  plan_status TEXT DEFAULT 'inactive', -- 
  plan_tier TEXT DEFAULT 'free',        -- 
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 2. PROFILES (User management linked to Orgs)
CREATE TABLE profiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id),
  org_id UUID REFERENCES organizations(id),
  full_name TEXT,
  role TEXT DEFAULT 'member', -- 'owner', 'admin', 'member' 
  daily_usage_limit INT DEFAULT 50, -- [cite: 8]
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 3. AI TASKS (The Orchestration Queue)
CREATE TABLE ai_tasks (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  org_id UUID REFERENCES organizations(id) NOT NULL,
  file_url TEXT NOT NULL,
  status TEXT DEFAULT 'PENDING', -- PENDING, EXTRACTING, ANALYZING, REVIEW_REQUIRED, COMPLETED 
  task_type TEXT, -- e.g., 'CAD_TAKEOFF', 'FINANCIAL_AUDIT' [cite: 17]
  raw_extraction JSONB, -- Step 1 results (Vision) [cite: 2, 8]
  final_analysis JSONB, -- Step 2/3 results (Math/Logic) [cite: 2, 17]
  is_verified BOOLEAN DEFAULT FALSE, -- Human-in-the-loop flag [cite: 2, 8]
  verified_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 4. AI USAGE (Cost & Quota Tracking)
CREATE TABLE ai_usage (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  org_id UUID REFERENCES organizations(id),
  month TEXT, -- e.g., '2026-03' [cite: 2]
  token_count INT DEFAULT 0,
  job_count INT DEFAULT 0,
  is_blocked BOOLEAN DEFAULT FALSE -- [cite: 2]
);

-- 5. ROW LEVEL SECURITY (RLS)
ALTER TABLE organizations ENABLE ROW LEVEL SECURITY;
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE ai_tasks ENABLE ROW LEVEL SECURITY;

-- Policy: Users can only see data belonging to their organization 
CREATE POLICY "Org-based access for AI tasks" ON ai_tasks
  FOR ALL USING (
    org_id IN (SELECT org_id FROM profiles WHERE id = auth.uid())
  );

CREATE POLICY "Users can view their own profile" ON profiles
  FOR SELECT USING (auth.uid() = id);
