import React, { useState } from 'react';
import { 
  X, 
  Check, 
  AlertTriangle, 
  Info, 
  Save, 
  Trash2, 
  ChevronRight,
  DatabaseZap
} from 'lucide-react';

/**
 * MOCK DATA: 
 * Representing the JSON extracted from a CAD floorplan or Financial Statement.
 */
const MOCK_EXTRACTED_DATA = [
  { id: 1, item: "Concrete Slab (Grade 30)", quantity: 450.5, unit: "m³", confidence: 0.94, status: 'unverified' },
  { id: 2, item: "Steel Reinforcement (12mm)", quantity: 12.4, unit: "tonnes", confidence: 0.82, status: 'unverified' },
  { id: 3, item: "Timber Formwork", quantity: 890, unit: "m²", confidence: 0.65, status: 'low_confidence' },
  { id: 4, item: "Vapor Barrier", quantity: 510, unit: "m²", confidence: 0.98, status: 'unverified' },
];

export default function ReviewResultsDrawer({ 
  isOpen, 
  onClose, 
  onSync 
}: { 
  isOpen: boolean; 
  onClose: () => void; 
  onSync: (data: any) => void;
}) {
  const [data, setData] = useState(MOCK_EXTRACTED_DATA);
  const [isSyncing, setIsSyncing] = useState(false);

  if (!isOpen) return null;

  const handleUpdate = (id: number, field: string, value: string | number) => {
    setData(prev => prev.map(item => item.id === id ? { ...item, [field]: value, status: 'modified' } : item));
  };

  const handleApproveAll = () => {
    setIsSyncing(true);
    // Simulate Sync to Supabase/ERP
    setTimeout(() => {
      setIsSyncing(false);
      onSync(data);
      onClose();
    }, 1500);
  };

  return (
    <div className="fixed inset-0 z-50 flex justify-end bg-black/40 backdrop-blur-sm">
      {/* DRAWER CONTAINER */}
      <div className="w-full max-w-2xl bg-background h-full shadow-2xl flex flex-col animate-in slide-in-from-right duration-300">
        
        {/* HEADER */}
        <div className="p-6 border-b flex items-center justify-between bg-muted/20">
          <div>
            <h2 className="text-xl font-bold flex items-center gap-2">
              Review AI Extraction
              <span className="text-xs font-normal bg-blue-100 text-blue-700 px-2 py-0.5 rounded-full">
                12 Items Found
              </span>
            </h2>
            <p className="text-sm text-muted-foreground mt-1">Verify and adjust quantities before syncing to ERP.</p>
          </div>
          <button onClick={onClose} className="p-2 hover:bg-muted rounded-full transition-colors">
            <X className="w-5 h-5" />
          </button>
        </div>

        {/* DATA LIST */}
        <div className="flex-1 overflow-y-auto p-6 space-y-4">
          <div className="flex items-center gap-2 p-3 bg-amber-50 border border-amber-200 rounded-lg text-amber-800 text-xs">
            <AlertTriangle className="w-4 h-4 shrink-0" />
            <span>Items highlighted in orange have a confidence score below 75% and require manual measurement verification.</span>
          </div>

          <table className="w-full text-sm border-collapse">
            <thead>
              <tr className="text-left text-muted-foreground border-b">
                <th className="pb-3 font-medium pl-2">Material / Line Item</th>
                <th className="pb-3 font-medium text-center">Confidence</th>
                <th className="pb-3 font-medium text-right pr-2">Estimated Qty</th>
              </tr>
            </thead>
            <tbody className="divide-y">
              {data.map((row) => (
                <tr key={row.id} className={`group hover:bg-muted/30 transition-colors ${row.confidence < 0.75 ? 'bg-amber-50/30' : ''}`}>
                  <td className="py-4 pl-2">
                    <div className="font-semibold">{row.item}</div>
                    <div className="text-[10px] text-muted-foreground flex items-center gap-1 mt-0.5 uppercase tracking-tighter">
                      <Info className="w-3 h-3" /> Page 4, Section B-12
                    </div>
                  </td>
                  <td className="py-4 text-center">
                    <div className={`inline-block px-2 py-0.5 rounded text-[10px] font-bold ${
                      row.confidence > 0.9 ? 'bg-green-100 text-green-700' : 
                      row.confidence > 0.7 ? 'bg-blue-100 text-blue-700' : 
                      'bg-amber-100 text-amber-700'
                    }`}>
                      {(row.confidence * 100).toFixed(0)}%
                    </div>
                  </td>
                  <td className="py-4 text-right pr-2">
                    <div className="flex items-center justify-end gap-2">
                      <input 
                        type="number" 
                        value={row.quantity}
                        onChange={(e) => handleUpdate(row.id, 'quantity', parseFloat(e.target.value))}
                        className="w-20 bg-background border rounded px-2 py-1 text-right focus:ring-1 focus:ring-blue-500 outline-none"
                      />
                      <span className="text-muted-foreground w-8 text-left">{row.unit}</span>
                    </div>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>

        {/* FOOTER ACTIONS */}
        <div className="p-6 border-t bg-muted/10 flex items-center justify-between gap-4">
          <button 
            onClick={onClose}
            className="px-4 py-2 text-sm font-medium hover:bg-muted rounded-lg transition-colors"
          >
            Save Draft
          </button>
          
          <div className="flex gap-2">
            <button 
              onClick={handleApproveAll}
              disabled={isSyncing}
              className="px-6 py-2 bg-blue-600 hover:bg-blue-700 disabled:bg-blue-400 text-white rounded-lg font-semibold flex items-center gap-2 transition-all shadow-md active:scale-95"
            >
              {isSyncing ? (
                <>
                  <DatabaseZap className="w-4 h-4 animate-pulse" />
                  Syncing to ERP...
                </>
              ) : (
                <>
                  <Check className="w-4 h-4" />
                  Approve & Finalize Data
                </>
              )}
            </button>
          </div>
        </div>
      </div>
    </div>
  );
}
